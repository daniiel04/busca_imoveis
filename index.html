function handleDownload() { var urlValue = document.getElementById('urlInput').value.trim(); var passwordValue = document.getElementById('passwordInput').value.trim(); var resultElement = document.getElementById('result'); // Limpar mensagem anterior resultElement.innerHTML = ''; // Verificar se ambos os campos estão vazios if (urlValue === '' && passwordValue === '') { resultElement.innerHTML = 'Por favor, insira a URL de busca e a senha.'; resultElement.classList.add('error'); return; } // Verificar se apenas a senha foi preenchida if (urlValue === '' && passwordValue !== '') { resultElement.innerHTML = 'Por favor, insira a URL do Imovirtual.'; resultElement.classList.add('error'); return; } // Verificar se apenas a URL foi preenchida if (urlValue !== '' && passwordValue === '') { resultElement.innerHTML = 'Por favor, insira a senha para acessar.'; resultElement.classList.add('error'); return; } // Iniciar mensagem de progresso resultElement.innerHTML = 'Download em andamento, isto pode levar alguns minutos...'; resultElement.classList.remove('success'); resultElement.classList.remove('error'); resultElement.classList.add('progress'); // Verificar se o URL é do Imovirtual if (!urlValue.startsWith('https://www.imovirtual.com/')) { resultElement.innerHTML = 'URL inválida. Por favor, forneça uma URL do Imovirtual.'; resultElement.classList.remove('progress'); resultElement.classList.add('error'); return; } // Enviar requisição para o servidor para download var xhr = new XMLHttpRequest(); xhr.open('GET', `/download?url=${encodeURIComponent(urlValue)}&password=${encodeURIComponent(passwordValue)}`, true); xhr.responseType = 'blob'; xhr.onload = function() { if (xhr.status === 200) { var blob = xhr.response; var link = document.createElement('a'); link.href = window.URL.createObjectURL(blob); link.download = 'dados_imoveis.xlsx'; document.body.appendChild(link); link.click(); document.body.removeChild(link); resultElement.innerHTML = 'Download concluído!'; resultElement.classList.remove('progress'); resultElement.classList.add('success'); } else if (xhr.status === 401) { resultElement.innerHTML = 'Senha incorreta. Tente novamente.'; resultElement.classList.remove('progress'); resultElement.classList.add('error'); } else { resultElement.innerHTML = 'Erro ao processar sua solicitação.'; resultElement.classList.remove('progress'); resultElement.classList.add('error'); } }; xhr.onerror = function() { resultElement.innerHTML = 'Erro ao conectar com o servidor.'; resultElement.classList.remove('progress'); resultElement.classList.add('error'); }; xhr.send(); }
